	    public int updateHTTPS(String name, boolean https){
        String SQL = "UPDATE municipalities SET HTTPS=" + 
            (https?"1":"0") + 
            " WHERE Name='" + name + "'";
        Statement stm = null;
        int rows = -1;
        try{
            stm  = con.createStatement();
            rows = stm.executeUpdate(SQL);
        }catch(Exception e){
            System.err.println("Exception updating https for " + name 
                               + " " + e.getMessage());
        }finally{
            closeIt(stm);
            return rows;
        }
    }
